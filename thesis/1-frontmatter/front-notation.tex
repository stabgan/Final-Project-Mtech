\nomenclature{$x_i$}{The $i$-th clinical note (free-text discharge summary)}
\nomenclature{$Y_i$}{The set of ICD codes associated with the $i$-th clinical note}
\nomenclature{$\mathcal{L}$}{The complete label space of ICD codes}
\nomenclature{$f(\cdot)$}{The learned multi-label classification function mapping text to sets of labels}
\nomenclature{$\theta$}{Trainable parameters of the classification model}
\nomenclature{$D$}{Overall dataset of paired clinical notes and labels}
\nomenclature{$\tilde{x}_j$}{Synthetic clinical note generated for augmentation}
\nomenclature{$P(Y \mid x;\theta)$}{Predicted probability of codes $Y$ given note $x$}
\nomenclature{$\mathrm{BCE}$}{Binary Cross Entropy, a common loss function for multi-label learning}
\nomenclature{$\mathrm{F1}$}{Harmonic mean of Precision and Recall, used for model evaluation}
\nomenclature{$\alpha$}{A learning rate or weight factor in optimization algorithms}
